<script>
  let quiz = [];
  let current = 0;
  let score = 0;
  let total = 0;
  let answered = [];

  const questionEl = document.getElementById("question");
  const choicesEl = document.getElementById("choices");
  const startBtn = document.getElementById("startBtn");
  const nextBtn = document.getElementById("nextBtn");
  const prevBtn = document.getElementById("prevBtn");
  const finishBtn = document.getElementById("finishBtn");
  const resultEl = document.getElementById("result");
  const scoreDisplay = document.getElementById("scoreDisplay");
  const setupDiv = document.getElementById("setup");

  const btnEasy = document.getElementById("btnEasy");
  const btnHard = document.getElementById("btnHard");

  // Zorluk seçimi butonları
  btnEasy.onclick = () => {
    btnEasy.classList.add("selected");
    btnHard.classList.remove("selected");
  };
  btnHard.onclick = () => {
    btnHard.classList.add("selected");
    btnEasy.classList.remove("selected");
  };

  startBtn.onclick = () => {
    const numQuestions = parseInt(document.getElementById("numQuestions").value);
    const difficulty = btnEasy.classList.contains("selected") ? "easy" : "hard";

    if (isNaN(numQuestions) || numQuestions < 1) {
      alert("Lütfen geçerli bir soru sayısı giriniz.");
      return;
    }

    const jsonFile = difficulty === "easy" ? "quiz-data.json" : "quiz-data_hard.json";

    fetch(jsonFile)
      .then(response => response.json())
      .then(data => {
        if (numQuestions > data.length) {
          alert(`Veritabanında sadece ${data.length} soru var, maksimum bu kadar soru sorulabilir.`);
          return;
        }

        quiz = getRandomQuestions(data, numQuestions);
        current = 0;
        score = 0;
        total = quiz.length;
        answered = new Array(total).fill(null);

        setupDiv.style.display = "none";
        questionEl.style.display = "block";
        choicesEl.style.display = "block";
        nextBtn.style.display = "inline-block";
        prevBtn.style.display = "inline-block";
        finishBtn.style.display = "block";
        document.querySelector(".controls").style.display = "flex";
        resultEl.textContent = "";
        scoreDisplay.textContent = "";

        showQuestion();
        updateButtons();
      });
  };

  nextBtn.onclick = () => {
    if (current < total - 1) {
      current++;
      showQuestion();
      updateButtons();
    }
  };
  prevBtn.onclick = () => {
    if (current > 0) {
      current--;
      showQuestion();
      updateButtons();
    }
  };

  finishBtn.onclick = () => {
    let correctCount = 0;
    for (let i = 0; i < total; i++) {
      if (answered[i] === quiz[i].answer) {
        correctCount++;
      }
    }
    scoreDisplay.textContent = `Doğru sayısı: ${correctCount} / ${total}`;
    resultEl.textContent = "Test tamamlandı! Tebrikler.";
    // Test bitti, butonları pasifleştirelim
    nextBtn.disabled = true;
    prevBtn.disabled = true;
    finishBtn.disabled = true;
    choicesEl.querySelectorAll("button").forEach(b => b.disabled = true);
  };

  function showQuestion() {
    questionEl.textContent = `Soru ${current + 1}: "${quiz[current].word}" kelimesinin anlamı nedir?`;
    choicesEl.innerHTML = "";
    quiz[current].choices.forEach(choice => {
      const btn = document.createElement("button");
      btn.textContent = choice;
      btn.onclick = () => {
        answered[current] = choice;
        updateScore();
        showQuestion();  // yenile, böylece işaretli cevap kalır
      };
      // Daha önce cevaplanmışsa buton rengi göster
      if (answered[current] !== null) {
        if (choice === quiz[current].answer) {
          btn.classList.add("correct");
        }
        if (choice === answered[current] && answered[current] !== quiz[current].answer) {
          btn.classList.add("wrong");
        }
        btn.disabled = true;
      }
      choicesEl.appendChild(btn);
    });
  }

  function updateScore() {
    let correctCount = 0;
    for (let i = 0; i < total; i++) {
      if (answered[i] === quiz[i].answer) correctCount++;
    }
    scoreDisplay.textContent = `Doğru sayısı: ${correctCount} / ${total}`;
  }

  function updateButtons() {
    prevBtn.disabled = (current === 0);
    nextBtn.disabled = (current === total - 1);
  }

  function getRandomQuestions(arr, num) {
    const shuffled = arr.slice().sort(() => 0.5 - Math.random());
    return shuffled.slice(0, num);
  }
</script>
